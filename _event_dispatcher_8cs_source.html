149  public HttpResponseMessage Dispatch(Event e, List<Subscription> subscriptions = null)
150  {
151  Log.Debug($"EventDispatcher.Dispatch: Dispatching event with name '{e.Name}'");
152 
153  if(subscriptions == null && eventSubscriptions.ContainsKey(e.Name)){
154  subscriptions = eventSubscriptions[e.Name];
155  }
156 
157  foreach (Subscription subscription in subscriptions.Where(x => !x.Synchronous))
158  {
159  Log.Debug($"EventDispatcher.Dispatch: Enqueuing item for subscriber '{subscription.Subscriber.Name}', for event '{e.Name}'");
160  QueueItem queueItem = new QueueItem()
161  {
162  Guid = Guid.NewGuid(),
163  Event = e,
164  Subscription = subscription,
165  Timestamp = DateTime.Now
166  };
167 
168  queue.Add(queueItem);
169  }
170  // move to last because the return will break the other Queue items
171  Subscription synchronousSubscription = subscriptions.FirstOrDefault(s => s.Synchronous);
172 
173  if (synchronousSubscription != null)
174  {
175  Log.Debug($"EventDispatcher.Dispatch: Executing Synchronous call for Event: '{e.Name}'");
176  return synchronousSubscription.SendEvent(e).Result;
177  }
178  // if there is no synchronous event then just return an empty response with 200 status code
179 
180  return new HttpResponseMessage(System.Net.HttpStatusCode.OK);
